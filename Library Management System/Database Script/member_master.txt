create table member_master(
member_id number GENERATED by default on null as IDENTITY,
member_type varchar(20),
member_name varchar(30),
book_issue_limit number,
total_book_issued number,
CONSTRAINTS member_master_pkey_constraint PRIMARY KEY(member_id),
CONSTRAINT  member_type_constraint CHECK(member_type IN('student','faculty'))
);
