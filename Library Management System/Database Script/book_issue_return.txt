create table book_issue_return(
issue_serial_number number(10),
book_code number,
member_id number,
date_of_issue date DEFAULT sysdate,
date_of_return date DEFAULT sysdate,
CONSTRAINTS book_issue_pkey_constraint PRIMARY KEY(issue_serial_number),
CONSTRAINTS book_issue_bcode_fkey FOREIGN KEY(book_code) REFERENCES book_master(book_code),
CONSTRAINTS book_issue_mcode_fkey FOREIGN KEY(member_id) REFERENCES member_master(member_id)
);
