create table student_faculty_master(
student_faculty_id varchar(20),
member_id number,
name varchar(30),
book_limit  number,
total_book_issued number,
CONSTRAINTS student_faculty_pkey_constraint PRIMARY KEY(student_faculty_id),
CONSTRAINTS member_id_fkey_constraint FOREIGN KEY(member_id) REFERENCES member_master(member_id),
CONSTRAINTS total_book_issued_constraint CHECK(total_book_issued<=book_limit),
CONSTRAINT pkey_like_constraint CHECK(student_faculty_id LIKE 'M%') 
);
